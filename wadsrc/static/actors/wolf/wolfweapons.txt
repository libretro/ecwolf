actor WolfWeapon : Weapon
{
	weapon.yadjust 16
}

actor Knife : WolfWeapon
{
	states
	{
		Ready:
			KNIF A 1 A_WeaponReady
			loop
		Fire:
			KNIF B 3
			KNIF C 3 A_Knife
			KNIF DE 3
			goto Ready
	}
}

actor Pistol : WolfWeapon
{
	states
	{
		Select:
			PISG A 1 A_Raise
			loop
		Deselect:
			PISG A 1 A_Lower
			loop
		Ready:
			PISG A 1 A_WeaponReady
			loop
		Fire:
			PISG B 3
			PISG C 3 A_Pistol
			PISG DE 3
			goto Ready
	}
}

actor MachineGun : WolfWeapon 60
{
	inventory.pickupsound "weapon/machine/pickup"
	states
	{
		Spawn:
			MGUN A -1
			stop
		Ready:
			MCHG A 1 A_WeaponReady
			loop
		Fire:
			MCHG B 3
		Hold:
			MCHG C 3 A_MachineGun
			MCHG D 3
			MCHG E 3 A_ReFire
			goto Spawn
	}
}

actor GatlingGun : WolfWeapon 61
{
	inventory.pickupsound "weapon/gatling/pickup"
	states
	{
		Spawn:
			CGUN A -1
			stop
		Ready:
			CHGG A 1 A_WeaponReady
			loop
		Fire:
			CHGG B 3
		Hold:
			CHGG CD 3 A_Chaingun
			CHGG E 3 A_ReFire
			goto Spawn
	}
}
