actor Hans 16
{
	radius 42
	points 5000
	health 850, 950, 1050, 1200
	speed 1, 3
	attacksound "hans/attack"
	deathsound "hans/death"
	seesound "hans/sight"
	dropitem "GoldKey"
	MONSTER
	+AMBUSH
	states
	{
		Spawn:
			HANS A -1 NOP T_Stand
			stop
		See:
			HANS A 5 NOP T_Chase
			HANS A 1.5
			HANS B 4 NOP T_Chase
			HANS C 5 NOP T_Chase
			HANS C 1.5
			HANS D 4 NOP T_Chase
			loop
		Missile:
			HANS E 15
			HANS F 5
			HANS GFGFGE 5 A_WolfAttack(0, "*", 0.667)
			goto See
		Death:
			HANS H 7.5 A_Fall
			HANS I 7.5 A_DeathScream
			HANS J 7.5
			HANS K -1
			stop
	}
}
